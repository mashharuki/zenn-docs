    const beforeBalance = await publicClient.getBalance({
      address: contractAddress as `0x${string}`,
    });

    // call withdraw method
    const hash = await lock.write.withdraw();

    const afterBalance = await publicClient.getBalance({
      address: contractAddress as `0x${string}`,
